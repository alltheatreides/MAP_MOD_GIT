add_namespace = ATOM

#setting up of the initial budget for majors
country_event = {
	id = ATOM.1
	title = ATOM.1.t
	desc = ATOM.1.desc
	
	is_triggered_only = yes
	hidden = yes

	immediate = {
		every_country = {
            #country_event = { id = newyears.1 days = 359 }
        }
		FRA = {
			set_variable = {
				var = budget
				value = 146
			}
		}
		GER = {
			set_variable = {
				var = budget
				value = 246
			}
		}
	}

}

#daily budget growth for majors
country_event = {
	id = ATOM.2
	title = ATOM.2.t
	desc = ATOM.2.desc

	is_triggered_only = yes
	hidden = yes

	trigger = {
		#might add a trigger for the first year of growth ? So second year of growth is different ?
		#best to do a monthly thing
	}

	immediate = {
		if = {
			limit = { tag = FRA	}
			add_to_variable = {
				var = budget
				value = 146
			}
		}
		if = {
			limit = { tag = GER	}
			add_to_variable = {
				var = budget
				value = 246
			}
		}
	}
}
