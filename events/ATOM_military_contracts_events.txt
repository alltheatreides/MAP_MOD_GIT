	id = ATOM_military_contracts.1
	title = ATOM_military_contracts.1.t
	desc = {

		trigger = {
			check_variable = { inf_eq_order_amount > 1 }
		}
	}
	desc = {
		text = ATOM_military_contracts.1.desc.b
		trigger = {
			check_variable = { h_inf_eq_order_amount > 1 }
		}
	}
	desc = {
		text = ATOM_military_contracts.1.desc.c
		trigger = {
			check_variable = { h_inf_eq_order_amount < 1 }
			check_variable = { inf_eq_order_amount < 1 }
		}
	}
	desc = {
		text = ATOM_military_contracts.1.desc.d
		trigger = {
			check_variable = { support_eq_order_amount > 1 }
		}
	}
	
	is_triggered_only = yes

	immediate = {
		
	}

	option = {
		name = ATOM_military_contracts.1.a
		trigger = {
			check_variable = { inf_eq_order_amount > 1 }
		}
		hidden_effect = {
			#Make some conditions for the time it takes to fulfull the order depending on the size of the order
			country_event = {
				id = ATOM_military_contracts.2 
				days = 150
			}
		}
	}

	option = {
		name = ATOM_military_contracts.1.b
		trigger = {
			check_variable = { h_inf_eq_order_amount > 1 }
		}
		hidden_effect = {
			country_event = {
				id = ATOM_military_contracts.2 
				days = 1
			}
		}
	}

	option = {
		name = ATOM_military_contracts.1.e
		trigger = {
			check_variable = { support_eq_order_amount > 1 }
		}
		hidden_effect = {
			country_event = {
				id = ATOM_military_contracts.2 
				days = 1
			}
		}
	}

	option = {
		name = ATOM_military_contracts.1.c
		trigger = {
			check_variable = { h_inf_eq_order_amount < 1 }
			check_variable = { inf_eq_order_amount < 1 }
			check_variable = { support_eq_order_amount < 1 }
		}
	}

	option = {
		name = ATOM_military_contracts.1.d
		trigger = {
			OR = {
				check_variable = { h_inf_eq_order_amount > 1 }
				check_variable = { inf_eq_order_amount > 1 }
				check_variable = { support_eq_order_amount > 1 }
			}
		}
	}

}
